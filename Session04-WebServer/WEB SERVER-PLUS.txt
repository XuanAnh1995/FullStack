1. JSX = JAVA SCRIPT XML 

-JSX LÀ NGÔN NGỮ JS MỞ RỘNG, CHO PHÉP VIẾT HTML TRONG CODE JS, TRONG CODE JS NHÚNG THÊM CÁC TAG HTML, TRONG HTML LẠI LỒNG CODE JS

const msg = <h1>Xin chào</h1>;

function BigMessage(){
	return <h1>Xin chào</h1>;
}

function BigMessageDetail() {
	return (
		<div>
			<h1>Xin chào</h1>
			<h2>My name is giáo.làng</h2>
		</div>
	);
}

function BigMessageDetail() {
	const yob = 2000;
	return (
		<>
			<h1>Xin chào</h1>
			<h2>My name is giáo.làng | Age: {2026 - yob}</h2>
		</>
	);
}
